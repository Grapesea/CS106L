```c++
// class.h

#ifndef CLASS_H
#define CLASS_H
#include<iostream>
#include<string>
class Juice{
    private:
        std::string name;
        std::string owner;
        const int sweetness;
        int remained_num;
        int price;
        bool is_valid_num(int num);
        
    public:
        Juice();
        Juice(const std::string name, std::string owner, int remained, int price);
        Juice(const Juice& j);
        ~Juice();
        int getData() const;
        int get_price() const;
        std::string get_owner_name() const;
        std::string get_name() const;
        void change_owner(const std::string new_owner);
        void set_price(int price);
        int  sell(int num); // sell num cups of Juice and return the money earned.
        void make(int num); // make num cups of Juice;
        void change_price(const int new_price);
};

#endif
```

getter函数后面放的`const`表示这个函数承诺不对class中的private参数进行修改，仅作取值.

`sandbox.cpp`不需要添加`#include "class.cpp"`，因为采用的指令是`g++ -std=c++20 main.cpp class.cpp -o main`，在编译阶段每个 `.cpp` 文件独立编译，`#include` 的作用是把文件内容直接复制粘贴进来，目的是让编译器知道函数/类的声明（即"这个东西长什么样"）。`class.h` 里有所有的声明，所以 `sandbox.cpp` 只需要 `#include "class.h"` 就够了.

等到链接阶段，编译完成之后链接器会将`.o`文件全部合并到一起，负责找到所有函数的实际实现.
